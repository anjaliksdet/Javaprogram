package LearningJava;

import javax.print.DocFlavor;
import java.security.KeyStore;

public class Ex_04_Literals {
    static void main() {
        //providing the literals
        int a = 65;
        //providing the constants
        final int b = 98;
        //using printf
        System.out.printf("the value of b = %d\n",b);
        //using println
        System.out.println("the value of b ="+b);
        //boolean literal
        boolean i = true;
        System.out.println(i);
        // using binary literal
        int binary = 0b1010;
        System.out.println(binary);
        //character litrals
        char w ='9';
        System.out.println(w);
        //Sting literals
        String o= "HELLO";
        System.out.println(o);
        //ch
        char ch= '@';
        System.out.println("the value provided is"+ch+"Anjali");


    }
}
