    static void main() {
//        Scanner sc= new Scanner(System.in);

        for (int i=0;i<=50;i++)
        {
            if (i%2==0){
                System.out.println("Even->" +i);
            }
            else {
                System.out.println("odd->"+ i);
            }
        }
    }
